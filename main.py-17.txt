# ─────────────────────────────────────────────────────────────────
# RATE LIMIT SYSTEM
# ─────────────────────────────────────────────────────────────────
GLOBAL_API_SEMAPHORE = asyncio.Semaphore(5)

async def rate_limited(coro_func: Callable, *args, delay: float = 0.6,
                       max_retries: int = 3, **kwargs) -> Any:
    attempts = 0
    while True:
        async with GLOBAL_API_SEMAPHORE:
            try:
                result = await coro_func(*args, **kwargs)
                if delay > 0: await asyncio.sleep(delay)
                return result
            except discord.HTTPException as exc:
                if exc.status == 429 and attempts < max_retries:
                    retry_after = 5.0
                    try: retry_after = float(exc.response.headers.get("Retry-After", 5))
                    except: pass
                    await asyncio.sleep(retry_after)
                    attempts += 1
                else: raise

class Database:
    CONFIG_CACHE_TTL = 300
    CONFIG_CACHE_MAXSIZE = 10000

    def __init__(self) -> None:
        mongo_url = os.getenv("MONGO_URL", "mongodb://localhost:27017")
        self.client = AsyncIOMotorClient(mongo_url, serverSelectionTimeoutMS=8000, maxPoolSize=50)
        self.db = self.client["ModForge"]
        self.config          = self.db["config"]
        self.whitelist       = self.db["whitelist"]
        self.data            = self.db["data"]
        self.tempactions     = self.db["tempactions"]
        self.cases           = self.db["cases"]
        self.counters        = self.db["counters"]
        self.message_archive = self.db["message_archive"]
        self.guild_events    = self.db["guild_events"]
        self.admin_audit_logs= self.db["admin_audit_logs"]
        self.mod_notes       = self.db["mod_notes"]
        self.backups         = self.db["backups"]
        self.gdpr_requests   = self.db["gdpr_requests"]
        self.voice_mutes     = self.db["voice_mutes"]
        self.invite_tracker  = self.db["invite_tracker"]
        self.api_tokens      = self.db["api_tokens"]
        self.stats_daily     = self.db["stats_daily"]
        self._config_cache: TTLCache = TTLCache(maxsize=self.CONFIG_CACHE_MAXSIZE, ttl=self.CONFIG_CACHE_TTL)
        self._whitelist_cache: TTLCache = TTLCache(maxsize=self.CONFIG_CACHE_MAXSIZE, ttl=self.CONFIG_CACHE_TTL)
        self._config_locks: Dict[int, asyncio.Lock] = {}
        self._whitelist_locks: Dict[int, asyncio.Lock] = {}

    def _config_lock(self, gid: int) -> asyncio.Lock:
        if gid not in self._config_locks: self._config_locks[gid] = asyncio.Lock()
        return self._config_locks[gid]

    def _whitelist_lock(self, gid: int) -> asyncio.Lock:
        if gid not in self._whitelist_locks: self._whitelist_locks[gid] = asyncio.Lock()
        return self._whitelist_locks[gid]

    @staticmethod
    def _ensure_defaults(cfg: dict) -> dict:
        for k, v in DEFAULT_CONFIG.items():
            if k not in cfg:
                cfg[k] = v.copy() if isinstance(v, (dict, list)) else v
            elif isinstance(v, dict) and isinstance(cfg.get(k), dict):
                for sk, sv in v.items():
                    cfg[k].setdefault(sk, sv)
        return cfg

    @staticmethod
    def _empty_whitelist() -> dict:
        return {"users": [], "roles": [], "channels": [], "bypass_antispam": [], "bypass_antinuke": []}

    def get_config(self, guild_id: int) -> dict:
        cached = self._config_cache.get(guild_id)
        if cached is not None: return cached
        default = self._ensure_defaults(DEFAULT_CONFIG.copy())
        default["_id"] = guild_id
        self._config_cache[guild_id] = default
        try:
            loop = asyncio.get_running_loop()
            loop.create_task(self.fetch_config(guild_id))
        except RuntimeError: pass
        return default

    async def fetch_config(self, guild_id: int) -> dict:
        async with self._config_lock(guild_id):
            try: data = await self.config.find_one({"_id": guild_id})
            except PyMongoError as e: log.error(f"fetch_config: {e}"); data = None
            if not data:
                data = DEFAULT_CONFIG.copy(); data["_id"] = guild_id
                try: await self.config.insert_one(data)
                except PyMongoError: pass
            data = self._ensure_defaults(data)
            self._config_cache[guild_id] = data
            return data

    async def set_config(self, guild_id: int, cfg: dict) -> None:
        cfg["_id"] = guild_id
        try: await self.config.update_one({"_id": guild_id}, {"$set": cfg}, upsert=True)
        except PyMongoError as e: log.error(f"set_config: {e}")
        self._config_cache[guild_id] = self._ensure_defaults(cfg)

    async def update_module(self, guild_id: int, module: str, key: str, value: Any) -> None:
        cfg = self._config_cache.get(guild_id) or await self.fetch_config(guild_id)
        cfg = dict(cfg); cfg.setdefault(module, {})
        if isinstance(cfg[module], dict): cfg[module] = {**cfg[module], key: value}
        else: cfg[module] = value
        await self.set_config(guild_id, cfg)

    async def aget_config(self, guild_id: int) -> dict:
        cached = self._config_cache.get(guild_id)
        if cached is not None: return cached
        return await self.fetch_config(guild_id)

    async def aset_config(self, guild_id: int, cfg: dict) -> None:
        await self.set_config(guild_id, cfg)

    def get_whitelist(self, guild_id: int) -> dict:
        cached = self._whitelist_cache.get(guild_id)
        if cached is not None: return cached
        empty = self._empty_whitelist()
        self._whitelist_cache[guild_id] = empty
        try:
            loop = asyncio.get_running_loop()
            loop.create_task(self.fetch_whitelist(guild_id))
        except RuntimeError: pass
        return empty

    async def fetch_whitelist(self, guild_id: int) -> dict:
        async with self._whitelist_lock(guild_id):
            try: data = await self.whitelist.find_one({"_id": guild_id})
            except PyMongoError: data = None
            if not data: data = self._empty_whitelist()
            for k in ("users","roles","channels","bypass_antispam","bypass_antinuke"):
                data.setdefault(k, [])
            self._whitelist_cache[guild_id] = data
            return data

    async def set_whitelist(self, guild_id: int, whitelist: dict) -> None:
        whitelist["_id"] = guild_id
        try: await self.whitelist.update_one({"_id": guild_id}, {"$set": whitelist}, upsert=True)
        except PyMongoError: pass
        self._whitelist_cache[guild_id] = whitelist

    async def aget_whitelist(self, guild_id: int) -> dict:
        cached = self._whitelist_cache.get(guild_id)
        if cached is not None: return cached
        return await self.fetch_whitelist(guild_id)

    async def aadd_whitelist(self, guild_id: int, category: str, entry_id: int) -> None:
        data = await self.aget_whitelist(guild_id); data = dict(data)
        data.setdefault(category, [])
        if entry_id not in data[category]: data[category] = data[category] + [entry_id]
        await self.set_whitelist(guild_id, data)

    async def aremove_whitelist(self, guild_id: int, category: str, entry_id: int) -> None:
        data = await self.aget_whitelist(guild_id); data = dict(data)
        if category in data: data[category] = [x for x in data[category] if x != entry_id]
        await self.set_whitelist(guild_id, data)

    async def aadd_warning(self, guild_id: int, user_id: int, reason: str, mod_id: int) -> int:
        try:
            await self.data.insert_one({"type": "warning", "guild_id": guild_id, "user_id": user_id,
                                         "reason": reason, "mod_id": mod_id, "timestamp": datetime.datetime.utcnow()})
        except PyMongoError: pass
        return len(await self.aget_warnings(guild_id, user_id))

    async def aget_warnings(self, guild_id: int, user_id: int) -> List[dict]:

class ActivityStream:
    def __init__(self, maxlen: int = 500) -> None:
        self.events: deque = deque(maxlen=maxlen)
        self._lock = threading.Lock()

    def push(self, kind: str, text: str, guild_id: Optional[int] = None,
             guild_name: Optional[str] = None, user_id: Optional[int] = None,
             user_name: Optional[str] = None, extra: Optional[dict] = None) -> None:
        evt = {"kind": kind, "text": text, "guild_id": guild_id, "guild_name": guild_name,
               "user_id": user_id, "user_name": user_name, "extra": extra or {},
               "ts": datetime.datetime.utcnow().isoformat() + "Z"}
        with self._lock: self.events.append(evt)

    def snapshot(self, limit: int = 200) -> List[dict]:
        with self._lock: data = list(self.events)
        return data[-limit:][::-1]

def create_embed(title: str, description: str = "", color: int = COLOR_PRIMARY,
                 fields: Optional[List[tuple]] = None, thumbnail: Optional[str] = None,
                 image: Optional[str] = None, user: Optional[Any] = None,
                 guild_cfg: Optional[dict] = None) -> discord.Embed:
    log_cfg = (guild_cfg or {}).get("log_embed", {})
    footer_text = log_cfg.get("footer_text", FOOTER_TEXT)
    footer_icon = log_cfg.get("footer_icon", FOOTER_ICON)
    embed = discord.Embed(title=title, description=description, color=color,
                          timestamp=datetime.datetime.now(datetime.timezone.utc))
    embed.set_footer(text=footer_text, icon_url=footer_icon)
    if fields:
        for name, value, inline in fields:
            embed.add_field(name=str(name), value=str(value), inline=bool(inline))
    if thumbnail: embed.set_thumbnail(url=thumbnail)
    if image: embed.set_image(url=image)
    if user: embed.set_author(name=f"{user.name}", icon_url=user.display_avatar.url)
    return embed

def parse_duration(text: str) -> Optional[int]:
    if not text: return None
    m = re.match(r"^(\d+)\s*([smhdw]?)$", text.strip().lower())
    if not m: return None
    val = int(m.group(1))
    return val * {"s":1,"m":60,"h":3600,"d":86400,"w":604800}.get(m.group(2) or "s", 1)

def _format_duration(seconds: Optional[int]) -> str:
    if not seconds: return "—"
    if seconds < 60: return f"{seconds}s"
    if seconds < 3600: return f"{seconds//60}m"
    if seconds < 86400: return f"{seconds//3600}h"
    return f"{seconds//86400}d"

def can_moderate(actor: discord.Member, target: discord.Member, bot_member: discord.Member) -> Tuple[bool, str]:
    if target.id == actor.guild.owner_id: return False, "Server-Owner kann nicht moderiert werden."
    if target.id == bot_member.id: return False, "Ich kann mich nicht selbst moderieren."
    if target.id == actor.id: return False, "Du kannst dich nicht selbst moderieren."
    if actor.id != actor.guild.owner_id and actor.top_role <= target.top_role:
        return False, "Deine höchste Rolle ist nicht über der des Ziels."
    if bot_member.top_role <= target.top_role: return False, "Meine höchste Rolle ist nicht über der des Ziels."
    return True, ""

def generate_captcha(difficulty: str = "medium") -> Tuple[str, "_io.BytesIO"]:
    length = 4 if difficulty=="easy" else (6 if difficulty=="medium" else 8)
    code = ''.join(random.choices(string.ascii_uppercase + string.digits, k=length))
    if not _PIL_AVAILABLE:
        return code, _io.BytesIO(b"")
    width, height = 280, 100
    img = Image.new('RGB', (width, height), (255, 255, 255))
    d = ImageDraw.Draw(img)
    try: font = ImageFont.truetype("arial.ttf", 40)
    except: font = ImageFont.load_default()
    d.text((width//4, height//3), code, fill=(0,0,0), font=font)
    for _ in range(300 if difficulty=="medium" else (100 if difficulty=="easy" else 600)):
        d.point((random.randint(0,width), random.randint(0,height)),
                fill=(random.randint(0,255),random.randint(0,255),random.randint(0,255)))
    buf = _io.BytesIO(); img.save(buf, format='PNG'); buf.seek(0)
    return code, buf

class CaptchaModal(discord.ui.Modal, title="Verifizierung"):
    answer = discord.ui.TextInput(label="Code aus dem Bild eingeben",
                                  placeholder="CODE123", min_length=4, max_length=8)
    def __init__(self, correct_code: str, bot_ref: Any, add_roles: List[int], remove_roles: List[int]) -> None:
        super().__init__(); self.correct_code=correct_code; self.bot=bot_ref
        self.add_roles=add_roles; self.remove_roles=remove_roles

    async def on_submit(self, interaction: discord.Interaction) -> None:
        if self.answer.value.upper() == self.correct_code:
            try:
                for r_id in self.add_roles:
                    role = interaction.guild.get_role(r_id)
                    if role: await interaction.user.add_roles(role, reason="Verifiziert")
                for r_id in self.remove_roles:
                    role = interaction.guild.get_role(r_id)
                    if role: await interaction.user.remove_roles(role, reason="Verifiziert")
                await interaction.response.send_message(f"{E.OK} Verifizierung erfolgreich!", ephemeral=True)
            except discord.Forbidden:
                await interaction.response.send_message(f"{E.FAIL} Mir fehlen Rechte.", ephemeral=True)
        else:
            await interaction.response.send_message(f"{E.FAIL} Falscher Code! Versuche es erneut.", ephemeral=True)

class TicketView(discord.ui.View):
    def __init__(self, bot_ref: Any) -> None:
        super().__init__(timeout=None); self.bot = bot_ref

    @discord.ui.button(label="Ticket öffnen", style=discord.ButtonStyle.blurple,
                       custom_id="modforge:open_ticket", emoji=E.TICKET)
    async def open_ticket(self, interaction: discord.Interaction, button: discord.ui.Button) -> None:
        cfg = self.bot.db.get_config(interaction.guild.id)["ticket_system"]
        category = interaction.guild.get_channel(cfg["category_id"]) if cfg.get("category_id") else None
        overwrites = {
            interaction.guild.default_role: discord.PermissionOverwrite(read_messages=False),
            interaction.user: discord.PermissionOverwrite(read_messages=True, send_messages=True),
            interaction.guild.me: discord.PermissionOverwrite(read_messages=True, send_messages=True),
        }
        try:
            ch = await interaction.guild.create_text_channel(f"ticket-{interaction.user.name}",
                                                              category=category, overwrites=overwrites)
        except discord.Forbidden:
            await interaction.response.send_message(f"{E.FAIL} Mir fehlen Rechte.", ephemeral=True); return
        await interaction.response.send_message(f"{E.TICKET_OK} {ch.mention}", ephemeral=True)
        await ch.send(embed=create_embed(f"{E.TICKET} Support-Ticket", f"Willkommen {interaction.user.mention}!", COLOR_INFO),
                      view=TicketCloseView(self.bot))

class ModForge(commands.Bot):
    def __init__(self) -> None:
        intents = discord.Intents.all()
        super().__init__(command_prefix="!", intents=intents, help_command=None)
        self.db = Database()
        self.tracker = Tracker()
        self.start_time: float = time.time()

    async def setup_hook(self) -> None:
        global BOT_REF; BOT_REF = self
        await self.db.ensure_indexes()
        self.add_view(VerifyView(self)); self.add_view(TicketView(self))
        self.add_view(TicketCloseView(self)); self.add_view(AppealActionView())
        self.add_view(VpnManualApproveView(bot_ref=self))
        self.cleanup_trackers.start(); self.tempaction_loop.start()
        self.restore_persistent_mutes.start(); self.expire_voice_mutes_loop.start()
        self.health_check_loop.start()

    async def on_ready(self) -> None:
        log.info(f"Bot bereit: {self.user} – {len(self.guilds)} Server")
        await self.tree.sync()
        await self.change_presence(activity=discord.Activity(type=discord.ActivityType.watching,
                                                              name=f"{len(self.guilds)} Server"))
        ACTIVITY.push("bot_ready", f"Bot gestartet: {self.user}")
        # Cache invites
        for guild in self.guilds:
            try:
                invites = await guild.invites()
                self.tracker.invite_cache[guild.id] = {inv.code: inv.uses for inv in invites}
            except: pass

    async def log_action(self, guild: discord.Guild, title: str, description: str,
                          color: int = COLOR_INFO, fields: Optional[List[tuple]] = None,
                          user: Optional[Any] = None, module: str = "default") -> None:
        cfg = self.db.get_config(guild.id)
        log_channels = cfg.get("log_channels", {}) or {}
        channel_id = log_channels.get(module) or log_channels.get("default") or cfg.get("log_channel")
        if not channel_id: return
        channel = guild.get_channel(int(channel_id))
        if not channel: return
        embed = create_embed(title, description, color, fields, user=user, guild_cfg=cfg)
        try: await channel.send(embed=embed)
        except: pass

    async def punish(self, member: discord.Member, action: str, reason: str,
                     duration: int = 60) -> Optional[int]:
        guild = member.guild
        cfg = self.db.get_config(guild.id)
        notif_cfg = cfg.get("notifications", {})
        case_id: Optional[int] = None
        try:
            if action == "warn":
                count = await self.db.aadd_warning(guild.id, member.id, reason, self.user.id)
                await self._check_warn_thresholds(member, count)
                case_id = await self.db.acreate_case(guild.id, member.id, self.user.id, "warn", reason)
                if notif_cfg.get("dm_on_warn"):
                    try: await member.send(embed=create_embed("⚠️ Verwarnung", f"Du wurdest auf **{guild.name}** verwarnt.\n**Grund:** {reason}", COLOR_WARNING))
                    except: pass
            elif action == "timeout":
                until = discord.utils.utcnow() + datetime.timedelta(seconds=duration)
                await member.timeout(until, reason=reason)
                await self.db.aadd_mute(guild.id, member.id, reason, self.user.id, duration)
                case_id = await self.db.acreate_case(guild.id, member.id, self.user.id, "timeout", reason, duration)
            elif action == "kick":
                if notif_cfg.get("dm_on_kick"):
                    try: await member.send(embed=create_embed("👢 Gekickt", f"Du wurdest von **{guild.name}** gekickt.\n**Grund:** {reason}", COLOR_WARNING))
                    except: pass
                await member.kick(reason=reason)
                case_id = await self.db.acreate_case(guild.id, member.id, self.user.id, "kick", reason)
            elif action == "ban":
                if notif_cfg.get("dm_on_ban"):
                    try: await member.send(embed=create_embed("🔨 Gebannt", f"Du wurdest von **{guild.name}** gebannt.\n**Grund:** {reason}", COLOR_DANGER))
                    except: pass
                pre_messages = await self.db.aget_user_messages(guild.id, member.id)
                await member.ban(reason=reason, delete_message_days=0)
                case_id = await self.db.acreate_case(guild.id, member.id, self.user.id, "ban", reason)
                if case_id and pre_messages:
                    await self.db.aattach_messages_to_case(guild.id, case_id,
                        [{"channel_id": m.get("channel_id"), "content": m.get("content"),
                          "timestamp": m["timestamp"].isoformat() + "Z"} for m in pre_messages])
        except discord.Forbidden: log.warning(f"Forbidden punish {member} {action}")
        except discord.HTTPException as e: log.warning(f"HTTP punish: {e}")
        return case_id

    async def _check_warn_thresholds(self, member: discord.Member, warn_count: int) -> None:
        cfg = self.db.get_config(member.guild.id)
        for threshold_str, action in sorted(cfg.get("warn_system",{}).get("thresholds",{}).items(), key=lambda x: int(x[0])):
            try:
                if warn_count == int(threshold_str):
                    await self.punish(member, action, f"Auto: {warn_count} Verwarnungen")
            except ValueError: continue

    def is_whitelisted(self, member: discord.Member, bypass_type: Optional[str] = None) -> bool:
        wl = self.db.get_whitelist(member.guild.id)
        if member.id in wl.get("users",[]): return True
        for role in member.roles:
            if role.id in wl.get("roles",[]): return True
        if bypass_type and member.id in wl.get(bypass_type,[]): return True
        return False

    @tasks.loop(minutes=5)
    async def cleanup_trackers(self) -> None:
        now = time.time()
        for gd in self.tracker.spam_tracker.values():
            for dq in gd.values():
                while dq and now - dq[0] > 60: dq.popleft()

    @tasks.loop(seconds=30)
    async def tempaction_loop(self) -> None:
        try: due = await self.db.aget_due_tempactions()
        except: return
        for entry in due:
            guild = self.get_guild(entry["guild_id"])
            if not guild: await self.db.adeactivate_tempaction(entry["_id"]); continue
            try:
                if entry["action"] == "tempban":
                    try: await guild.unban(discord.Object(id=entry["user_id"]), reason="Tempban abgelaufen")
                    except: pass
                elif entry["action"] == "tempmute":
                    member = guild.get_member(entry["user_id"])
                    if member:
                        try: await member.timeout(None, reason="Tempmute abgelaufen")
                        except: pass
                    await self.db.adeactivate_mute(guild.id, entry["user_id"])
            finally: await self.db.adeactivate_tempaction(entry["_id"])

    @tasks.loop(seconds=60)
    async def expire_voice_mutes_loop(self) -> None:
        try:
            for entry in await self.db.aget_due_voice_mutes():
                guild = self.get_guild(entry["guild_id"])
                if guild:
                    member = guild.get_member(entry["user_id"])
                    if member and member.voice:
                        try: await member.edit(mute=False, reason="Voice-Mute abgelaufen")
                        except: pass
                await self.db.adeactivate_voice_mute(entry["guild_id"], entry["user_id"])
        except: pass

    @tasks.loop(count=1)
    async def restore_persistent_mutes(self) -> None:
        await self.wait_until_ready()
        for entry in await self.db.aget_active_mutes():
            guild = self.get_guild(entry["guild_id"])
            if not guild: continue
            member = guild.get_member(entry["user_id"])
            if not member: continue
            end_time = entry.get("end_time")
            now = datetime.datetime.utcnow()
            if end_time and end_time <= now:
                try: await member.timeout(None)
                except: pass
                await self.db.adeactivate_mute(guild.id, member.id)
            elif end_time:
                try: await member.timeout(end_time)
                except: pass

    @tasks.loop(minutes=10)
    async def health_check_loop(self) -> None:
        await self.wait_until_ready()
        try: await self.db.client.admin.command("ping"); mongo_ok = True
        except: mongo_ok = False
        latency_ms = round(self.latency * 1000, 1)
        if not mongo_ok or latency_ms > 1000:
            msg = (f"⚠️ **ModForge Health-Warnung**\nMongoDB: {'✅' if mongo_ok else '❌'}\nLatenz: {latency_ms}ms")
            log.warning(msg)
            if HEALTH_WEBHOOK_URL and _REQUESTS_AVAILABLE:
                try: _requests.post(HEALTH_WEBHOOK_URL, json={"content": msg}, timeout=5)
                except: pass

        gdpr_dm_sessions[user_id] = {"step": "confirm"}
        await message.author.send(embed=create_embed(
            f"{E.GDPR} DSGVO-Datenlöschung",
            "Schreibe `BESTÄTIGEN` um alle deine Daten zu löschen, oder `SERVER <ID>` für einen Server.\n`ABBRUCH` zum Abbrechen.",
            COLOR_WARNING))
        return True
    if user_id in gdpr_dm_sessions:
        if content.upper() == "BESTÄTIGEN":
            del gdpr_dm_sessions[user_id]
            deleted = await bot.db.adelete_user_data(user_id)
            req_id = await bot.db.aadd_gdpr_request(user_id, None, "full_delete")
            await message.author.send(embed=create_embed(f"{E.OK} DSGVO-Löschung abgeschlossen",
                f"**{sum(deleted.values())}** Datensätze gelöscht. ID: `{req_id}`", COLOR_SUCCESS))
        elif content.upper() == "ABBRUCH":
            del gdpr_dm_sessions[user_id]
            await message.author.send(embed=create_embed("🚫 Abgebrochen","Keine Änderungen.", COLOR_INFO))
        elif content.upper().startswith("SERVER "):
            parts = content.split()
            if len(parts) >= 2 and parts[1].isdigit():
                gid = int(parts[1]); del gdpr_dm_sessions[user_id]
                deleted = await bot.db.adelete_user_data(user_id, gid)
                req_id = await bot.db.aadd_gdpr_request(user_id, gid, "guild_delete")
                await message.author.send(embed=create_embed(f"{E.OK} Gelöscht",
                    f"**{sum(deleted.values())}** Einträge für Server `{gid}` gelöscht. ID: `{req_id}`", COLOR_SUCCESS))
        return True
    # Appeal flow
    if user_id in appeal_sessions:
        session = appeal_sessions[user_id]
        if content.lower() == "!abbruch":
            del appeal_sessions[user_id]
            await message.author.send(embed=create_embed("🚫 Appeal abgebrochen","",COLOR_WARNING))
            return True
        step = session["step"]
        if step == -1:
            if content.lower() == "!start":
                session["step"] = 0
                await message.author.send(embed=create_embed("📋 Appeal gestartet",
                    "Beantworte 5 Fragen ehrlich.\n\n" + APPEAL_QUESTIONS[0], COLOR_PRIMARY))
        else:
            session["answers"][APPEAL_KEYS[step]] = content
            session["step"] += 1
            if session["step"] < len(APPEAL_QUESTIONS):
                await message.author.send(embed=create_embed("✅ Gespeichert", APPEAL_QUESTIONS[session["step"]], COLOR_INFO))
            else:
                await _submit_appeal(message.author, session)
                del appeal_sessions[user_id]
        return True
    return False

async def _submit_appeal(user: discord.User, session: dict) -> None:
    guild_id = session["guild_id"]
    guild = bot.get_guild(guild_id)
    if not guild: await user.send("❌ Server nicht gefunden."); return
    cfg = bot.db.get_config(guild_id)
    channel_id = cfg.get("appeal_log_channel")
    if not channel_id: await user.send("❌ Kein Appeal-Kanal konfiguriert."); return
    channel = guild.get_channel(int(channel_id))
    if not channel: return
    embed = create_embed("📋 Entbannungs-Antrag", "Ein gebannter Nutzer möchte entbannt werden.", COLOR_PRIMARY,
        [("👤 Nutzer",f"{user.mention} (`{user}`)",True),("🆔 ID",str(user.id),True),
         ("❓ Warum gebannt?",session["answers"].get("ban_grund","–"),False),
         ("✅ Fehler?",session["answers"].get("fehler_eingesehen","–"),True),
         ("📜 Statement",session["answers"].get("statement","–"),False),
         ("🤝 Regeln?",session["answers"].get("regeln_versprechen","–"),True)],
        thumbnail=user.display_avatar.url)
    embed.set_footer(text=f"{FOOTER_TEXT} | UserID: {user.id}", icon_url=FOOTER_ICON)
    view = AppealActionView(bot_ref=bot, appeal_data={"user_id": user.id, "guild_id": guild_id})
    await channel.send(embed=embed, view=view)
    await user.send(embed=create_embed("✅ Appeal eingereicht!",f"Dein Antrag wurde an **{guild.name}** gesendet.",COLOR_SUCCESS))

async def _activate_lockdown(guild: discord.Guild) -> None:
    for ch in guild.text_channels:
        try: await ch.set_permissions(guild.default_role, send_messages=False, reason="Lockdown")
        except: pass
        await asyncio.sleep(0.3)
    await bot.log_action(guild, f"{E.LOCK} Lockdown aktiviert", "Alle Kanäle gesperrt.", COLOR_DANGER, module="antiraid")

async def _deactivate_lockdown(guild: discord.Guild) -> None:
    for ch in guild.text_channels:
        try: await ch.set_permissions(guild.default_role, send_messages=None, reason="Lockdown aufgehoben")
        except: pass
        await asyncio.sleep(0.3)
    bot.tracker.lockdown_active[guild.id] = False
    await bot.log_action(guild, f"{E.UNLOCK} Lockdown aufgehoben", "Alle Kanäle entsperrt.", COLOR_SUCCESS, module="antiraid")

async def _nuke_check(guild: discord.Guild, executor_id: int, action: str) -> None:
    cfg = bot.db.get_config(guild.id)
    nuke_cfg = cfg.get("anti_nuke", {})
    if not nuke_cfg.get("enabled"): return
    now = time.time(); dq = bot.tracker.nuke_tracker[guild.id][executor_id]
    dq.append(now); bot.tracker.clean_old(dq, nuke_cfg.get("window", 10))
    if len(dq) >= nuke_cfg.get("threshold", 5):
        dq.clear()
        executor = guild.get_member(executor_id)
        if not executor or bot.is_whitelisted(executor): return
        if nuke_cfg.get("remove_roles"):
            try: await executor.edit(roles=[], reason="Anti-Nuke")
            except: pass
        await bot.punish(executor, nuke_cfg.get("punishment","ban"), f"Anti-Nuke: {action}")
        await bot.log_action(guild, f"{E.SCAM} ANTI-NUKE",
            f"**{executor.mention}** verdächtige Massenaktionen!\n**Aktion:** {action}",
            COLOR_DANGER, user=executor, module="antinuke")

async def _audit_actor(guild: discord.Guild, action: discord.AuditLogAction,
                       target_id: Optional[int] = None) -> Optional[discord.abc.User]:
    try:
        async for entry in guild.audit_logs(limit=3, action=action):
            if target_id is None or (hasattr(entry.target,"id") and entry.target.id == target_id):
                return entry.user
    except: pass
    return None

@bot.event
async def on_member_join(member: discord.Member) -> None:
    guild = member.guild
    cfg = bot.db.get_config(guild.id)
    ACTIVITY.push("join", f"{member} → {guild.name}", guild_id=guild.id, guild_name=guild.name)
    await bot.db._record_daily_stat(guild.id, "joins")

    # Invite tracker
    invite_cfg = cfg.get("invite_tracker", {})
    if invite_cfg.get("enabled"):
        try:
            new_invites = await guild.invites()
            old_cache = bot.tracker.invite_cache.get(guild.id, {})
            used_code = None; inviter_id = None
            for inv in new_invites:
                if old_cache.get(inv.code, 0) < inv.uses:
                    used_code = inv.code; inviter_id = inv.inviter.id if inv.inviter else None; break
            if used_code and inviter_id:
                await bot.db.arecord_join_via_invite(guild.id, member.id, inviter_id, used_code)
                await bot.log_action(guild, f"🔗 Invite-Join",
                    f"{member.mention} nutzte Invite `{used_code}` von <@{inviter_id}>",
                    COLOR_INFO, user=member, module="invites")
            bot.tracker.invite_cache[guild.id] = {inv.code: inv.uses for inv in new_invites}
        except: pass

    # Auto-Role
    auto_role_cfg = cfg.get("auto_role", {})
    if auto_role_cfg.get("enabled"):
        role_list = auto_role_cfg.get("bot_roles",[]) if member.bot else auto_role_cfg.get("roles",[])
        for role_id in role_list:
            role = guild.get_role(role_id)
            if role:
                try: await member.add_roles(role, reason="Auto-Role")
                except: pass

    # Welcome
    welcome_cfg = cfg.get("welcome_system", {})
    if welcome_cfg.get("enabled") and not member.bot:
        ch = guild.get_channel(welcome_cfg.get("channel_id",0)) if welcome_cfg.get("channel_id") else None
        if ch:
            def fmt(s):
                return (s or "").replace("{user}", member.mention).replace("{server}", guild.name).replace("{count}", str(guild.member_count))
            welcome_embed = discord.Embed(
                title=fmt(welcome_cfg.get("embed_title","Willkommen!")),
                description=fmt(welcome_cfg.get("embed_description","Schön, dass du da bist {user}!")),
                color=welcome_cfg.get("embed_color", COLOR_SUCCESS),
                timestamp=datetime.datetime.now(datetime.timezone.utc)
            )
            welcome_embed.set_footer(text=FOOTER_TEXT, icon_url=FOOTER_ICON)
            if welcome_cfg.get("show_avatar"): welcome_embed.set_thumbnail(url=member.display_avatar.url)
            if welcome_cfg.get("embed_image_url"): welcome_embed.set_image(url=welcome_cfg["embed_image_url"])
            if welcome_cfg.get("embed_thumbnail_url"): welcome_embed.set_thumbnail(url=welcome_cfg["embed_thumbnail_url"])
            for field in welcome_cfg.get("embed_fields", []):
                welcome_embed.add_field(name=field.get("name",""), value=fmt(field.get("value","")), inline=field.get("inline",False))
            try: await ch.send(content=fmt(welcome_cfg.get("message","")), embed=welcome_embed)
            except: pass

    if bot.is_whitelisted(member, "bypass_antinuke"):
        await bot.log_action(guild, f"{E.JOIN} Beigetreten", f"{member.mention}", COLOR_SUCCESS, user=member, module="members")
        return

    sec_level = cfg.get("security_level", 0)
    if sec_level >= 2:
        account_age = (datetime.datetime.utcnow() - member.created_at.replace(tzinfo=None)).days
        if account_age < 5:
            try: await member.kick(reason="Sicherheitsstufe 2: Account zu jung")
            except: pass
            return
    if sec_level >= 3:
        try: await member.ban(reason="Sicherheitsstufe 3: BLACKOUT")
        except: pass
        return

    raid_cfg = cfg.get("anti_raid", {})
    if raid_cfg.get("enabled"):
        now = time.time()
        dq = bot.tracker.raid_tracker[guild.id]; dq.append(now)
        bot.tracker.clean_old(dq, raid_cfg.get("window",20))
        account_age = (datetime.datetime.utcnow() - member.created_at.replace(tzinfo=None)).days
        if account_age < raid_cfg.get("min_account_age",7) and raid_cfg.get("auto_kick"):
            try: await member.kick(reason=f"Anti-Raid: Account zu jung ({account_age}d)")
            except: pass
            return
        if len(dq) >= raid_cfg.get("join_threshold",10):
            dq.clear()
            if not bot.tracker.lockdown_active[guild.id] and raid_cfg.get("lockdown"):
                bot.tracker.lockdown_active[guild.id] = True
                await _activate_lockdown(guild)

    await bot.log_action(guild, f"{E.JOIN} Beigetreten", f"{member.mention}", COLOR_SUCCESS, user=member, module="members")

            except: pass

@bot.event
async def on_message(message: discord.Message) -> None:
    if not message.guild:
        await handle_appeal_dm(message)
        return
    if message.author.bot:
        await bot.process_commands(message); return
    member = message.author; guild = message.guild
    cfg = bot.db.get_config(guild.id)
    archive_cfg = cfg.get("message_archive", {}) or {}
    if archive_cfg.get("enabled"):
        try:
            await bot.db.arecord_message(guild.id, message.channel.id, message.id,
                member.id, message.content or "", [a.url for a in message.attachments])
        except: pass
    if bot.is_whitelisted(member): await bot.process_commands(message); return
    if message.mentions or message.role_mentions:
        if cfg.get("anti_ghost_ping",{}).get("enabled"):
            bot.tracker.ghost_tracker[guild.id][message.id] = (
                member.id, [u.id for u in message.mentions], [r.id for r in message.role_mentions])
    if bot.is_whitelisted(member, "bypass_antispam"): await bot.process_commands(message); return
    # Anti-Spam
    spam_cfg = cfg.get("anti_spam", {})
    if spam_cfg.get("enabled"):
        now = time.time(); dq = bot.tracker.spam_tracker[guild.id][member.id]
        dq.append(now); bot.tracker.clean_old(dq, spam_cfg.get("msg_window",10))
        if len(dq) > spam_cfg.get("msg_limit",5):
            try: await message.delete()
            except: pass
            await bot.punish(member, spam_cfg.get("punishment","timeout"), "Anti-Spam: Nachrichtenflut",
                             spam_cfg.get("timeout_duration",30))
            await bot.log_action(guild, f"{E.SPAM} Anti-Spam", f"{member.mention} Nachrichtenflut.",
                                  COLOR_WARNING, user=member, module="antispam")
            await bot.process_commands(message); return
        content = message.content or ""
        if len(content) > 10:
            caps_ratio = sum(1 for c in content if c.isupper()) / max(len(content),1) * 100
            if caps_ratio > spam_cfg.get("caps_pct",70):
                try: await message.delete()
                except: pass
                await bot.log_action(guild, f"{E.CAPS} Anti-Spam: Caps", f"{member.mention}", COLOR_WARNING, user=member, module="antispam")
        emoji_count = len(re.findall(r"<a?:[a-zA-Z0-9_]+:\d+>|[\U0001F300-\U0001FAFF]", content))
        if emoji_count > spam_cfg.get("emoji_max",10):
            try: await message.delete()
            except: pass
            await bot.log_action(guild, f"{E.EMOJI} Anti-Spam: Emojis ({emoji_count})", f"{member.mention}", COLOR_WARNING, user=member, module="antispam")
        dup_list = bot.tracker.dup_tracker[guild.id][member.id]
        msg_hash = hashlib.md5((content).encode()).hexdigest()
        dup_list.append(msg_hash)
        if len(dup_list) > spam_cfg.get("duplicate_max",3): dup_list.pop(0)
        if len(dup_list) >= spam_cfg.get("duplicate_max",3) and len(set(dup_list)) == 1:
            try: await message.delete()
            except: pass
            await bot.log_action(guild, f"{E.DUP} Anti-Spam: Duplikat", f"{member.mention}", COLOR_WARNING, user=member, module="antispam")
    # Anti-Mention
    mention_cfg = cfg.get("anti_mention", {})
    if mention_cfg.get("enabled") and (message.mentions or message.role_mentions):
        mc = len(message.mentions) + len(message.role_mentions)
        now = time.time(); dq = bot.tracker.mention_tracker[guild.id][member.id]
        dq.append(now); bot.tracker.clean_old(dq, mention_cfg.get("window",10))
        if mc > mention_cfg.get("mention_limit",5):
            try: await message.delete()
            except: pass
            await bot.punish(member, mention_cfg.get("punishment","timeout"), "Anti-Mention", mention_cfg.get("timeout_duration",60))
            await bot.log_action(guild, f"{E.MENTION} Anti-Mention ({mc})", f"{member.mention}", COLOR_WARNING, user=member, module="antimention")
            await bot.process_commands(message); return
    # AutoMod
    automod_cfg = cfg.get("automod", {})
    if automod_cfg.get("enabled") and message.content:
        content = message.content; content_lower = content.lower()
        for bw in automod_cfg.get("bad_words",[]):
            if bw in content_lower:
                try: await message.delete()
                except: pass
                await bot.punish(member, automod_cfg.get("punishment","warn"), f"AutoMod: Bad-Word")
                await bot.log_action(guild, f"{E.BADWORD} AutoMod: Bad-Word", f"{member.mention}", COLOR_WARNING, user=member, module="automod")
                await bot.process_commands(message); return
        for pattern in automod_cfg.get("regex_rules",[]):
            try:
                if re.search(pattern, content, re.IGNORECASE):
                    try: await message.delete()
                    except: pass
                    await bot.punish(member, automod_cfg.get("punishment","warn"), "AutoMod: Regex")
                    await bot.log_action(guild, f"{E.AUTOMOD} AutoMod: Regex", f"{member.mention}", COLOR_WARNING, user=member, module="automod")
                    await bot.process_commands(message); return
            except re.error: pass
        if automod_cfg.get("invite_filter") and INVITE_REGEX.search(content):
            try: await message.delete()
            except: pass
            await bot.punish(member, automod_cfg.get("punishment","warn"), "AutoMod: Invite")
            await bot.log_action(guild, f"{E.AUTOMOD} AutoMod: Invite", f"{member.mention}", COLOR_WARNING, user=member, module="automod")
            await bot.process_commands(message); return
        if automod_cfg.get("link_filter"):
            urls = URL_REGEX.findall(content)
            allowed = automod_cfg.get("allowed_domains",[])
            for url in urls:
                domain = re.sub(r"https?://([^/\s]+).*", r"\1", url.lower())
                if not any(ad in domain for ad in allowed):
                    try: await message.delete()
                    except: pass
                    await bot.punish(member, automod_cfg.get("punishment","warn"), f"AutoMod: Link ({domain})")
                    await bot.log_action(guild, f"{E.AUTOMOD} AutoMod: Link ({domain})", f"{member.mention}", COLOR_WARNING, user=member, module="automod")
                    await bot.process_commands(message); return
        if automod_cfg.get("zalgo_filter") and ZALGO_REGEX.search(content):
            try: await message.delete()
            except: pass
            await bot.log_action(guild, f"{E.AUTOMOD} AutoMod: Zalgo", f"{member.mention}", COLOR_WARNING, user=member, module="automod")
        if automod_cfg.get("phishing_check"):
            for url in URL_REGEX.findall(content):
                if await check_phishing_url(url):
                    try: await message.delete()
                    except: pass
                    await bot.punish(member, "ban", "AutoMod: Phishing")
                    await bot.log_action(guild, f"{E.SCAM} AutoMod: Phishing", f"{member.mention}", COLOR_DANGER, user=member, module="automod")
                    await bot.process_commands(message); return
        scam_cfg = cfg.get("anti_scam", {})
        if scam_cfg.get("enabled"):
            for sd in SCAM_DOMAINS:
                if sd in content_lower:
                    try: await message.delete()
                    except: pass
                    await bot.punish(member, scam_cfg.get("punishment","ban"), f"Anti-Scam: {sd}")
                    await bot.log_action(guild, f"{E.SCAM} Anti-Scam: {sd}", f"{member.mention}", COLOR_DANGER, user=member, module="antiscam")
                    await bot.process_commands(message); return
        shortener_cfg = cfg.get("anti_url_shortener", {})
        if shortener_cfg.get("enabled"):
            for us in URL_SHORTENERS:
                if us in content_lower:
                    try: await message.delete()
                    except: pass
                    await bot.punish(member, shortener_cfg.get("punishment","warn"), f"Anti-Shortener: {us}")
                    await bot.log_action(guild, f"{E.URLSHORT} Anti-Shortener: {us}", f"{member.mention}", COLOR_WARNING, user=member, module="antishortener")
                    await bot.process_commands(message); return
    await bot.process_commands(message)

@bot.event
async def on_message_delete(message: discord.Message) -> None:
    if not message.guild or message.author.bot: return
    cfg = bot.db.get_config(message.guild.id)
    if cfg.get("message_archive",{}).get("enabled"):
        await bot.db.amark_message_deleted(message.guild.id, message.id)
    if cfg.get("anti_ghost_ping",{}).get("enabled"):

            author_id, user_ids, role_ids = ghost_data
            author = message.guild.get_member(author_id)
            if author:
                mu = " ".join(f"<@{uid}>" for uid in user_ids); mr = " ".join(f"<@&{rid}>" for rid in role_ids)
                await bot.log_action(message.guild, f"{E.GHOST} Ghost-Ping",
                    f"{author.mention} löschte Msg mit Erwähnungen\n**User:** {mu or '–'}\n**Rollen:** {mr or '–'}",
                    COLOR_WARNING, user=author, module="automod")

@bot.event
async def on_member_update(before: discord.Member, after: discord.Member) -> None:
    guild = after.guild
    if before.roles != after.roles:
        added = [r for r in after.roles if r not in before.roles]
        removed = [r for r in before.roles if r not in after.roles]
        fields = []
        if added: fields.append((f"{E.ROLE_ADD} Hinzugefügt"," ".join(r.mention for r in added),True))
        if removed: fields.append((f"{E.ROLE_DEL} Entfernt"," ".join(r.mention for r in removed),True))
        if fields:
            await bot.log_action(guild, f"{E.ROLE} Rollen geändert", f"{after.mention}", COLOR_INFO, fields, user=after, module="members")
    if before.nick != after.nick:
        await bot.log_action(guild, f"{E.NICK} Nick geändert",
            f"{after.mention}: `{before.nick or 'Standard'}` → `{after.nick or 'Standard'}`",
            COLOR_INFO, user=after, module="nicknames")

@bot.event
async def on_guild_role_update(before: discord.Role, after: discord.Role) -> None:
    diffs = []
    if before.name != after.name: diffs.append(("Name", f"`{before.name}` → `{after.name}`", False))
    if before.color != after.color: diffs.append(("Farbe", f"`{before.color}` → `{after.color}`", True))
    if before.permissions != after.permissions:
        b_p = {p for p,v in before.permissions if v}; a_p = {p for p,v in after.permissions if v}
        added_p = a_p - b_p; removed_p = b_p - a_p
        if added_p: diffs.append(("➕ Permissions", "`"+"`, `".join(sorted(added_p))[:900]+"`", False))
        if removed_p: diffs.append(("➖ Permissions", "`"+"`, `".join(sorted(removed_p))[:900]+"`", False))
    if diffs:
        await bot.log_action(after.guild, f"{E.ROLE} Rolle aktualisiert", f"{after.mention} (`{after.id}`)",
                              COLOR_INFO, diffs, module="roles")

def has_mod_perms() -> Callable:
    async def predicate(ctx: commands.Context) -> bool:
        if ctx.author.guild_permissions.manage_messages: return True
        cfg = bot.db.get_config(ctx.guild.id)
        mod_role_id = cfg.get("mod_role")
        if mod_role_id: return any(r.id == int(mod_role_id) for r in ctx.author.roles)
        return False
    return commands.check(predicate)

@app_commands.describe(member="Nutzer", grund="Grund", delete_messages="Nachrichten löschen (0-7 Tage)")
@app_commands.default_permissions(ban_members=True)
async def slash_ban(interaction: discord.Interaction, member: discord.Member,
                    grund: str = "Kein Grund", delete_messages: int = 0) -> None:
    ok, why = can_moderate(interaction.user, member, interaction.guild.me)
    if not ok:
        await interaction.response.send_message(embed=create_embed(f"{E.FAIL} Blockiert",why,COLOR_DANGER), ephemeral=True); return

              "timestamp":m["timestamp"].isoformat()+"Z"} for m in pre_messages])
    embed = create_embed(f"{E.BAN} Gebannt",f"{member.mention} gebannt.",COLOR_DANGER,
                         [("Grund",grund,False),("Case",f"#{case_id}",True),("Mod",interaction.user.mention,True)])
    await interaction.response.send_message(embed=embed)
    await bot.log_action(interaction.guild,f"{E.BAN} Ban",f"{member.mention} gebannt.",COLOR_DANGER,
                          [("Grund",grund,False),("Case",f"#{case_id}",True)],user=member,module="moderation")
    await bot.db.aadd_admin_audit(interaction.guild.id,interaction.user.id,str(interaction.user),"ban","moderation",f"Target:{member.id} Grund:{grund}")

backup_group = app_commands.Group(name="backup", description="Server-Backups verwalten")

async def _create_server_snapshot(guild: discord.Guild) -> dict:
    snapshot: Dict[str, Any] = {
        "name": guild.name, "icon_url": str(guild.icon.url) if guild.icon else None,
        "verification_level": guild.verification_level.value,
        "explicit_content_filter": guild.explicit_content_filter.value,
        "roles": [], "categories": [], "text_channels": [], "voice_channels": [],
        "created_at": datetime.datetime.utcnow().isoformat(),
    }
    for role in guild.roles:
        if not role.is_default():
            snapshot["roles"].append({"name": role.name, "color": role.color.value,
                                       "hoist": role.hoist, "mentionable": role.mentionable,
                                       "position": role.position, "permissions": role.permissions.value})
    for cat in guild.categories:
        snapshot["categories"].append({"name": cat.name, "position": cat.position})
    for ch in guild.text_channels:
        snapshot["text_channels"].append({"name": ch.name, "position": ch.position, "topic": ch.topic,
                                           "nsfw": ch.is_nsfw(), "slowmode_delay": ch.slowmode_delay,
                                           "category_name": ch.category.name if ch.category else None})
    for vc in guild.voice_channels:
        snapshot["voice_channels"].append({"name": vc.name, "position": vc.position,
                                            "bitrate": vc.bitrate, "user_limit": vc.user_limit,
                                            "category_name": vc.category.name if vc.category else None})
    return snapshot

@backup_group.command(name="restore", description="Stellt ein Backup wieder her")
@app_commands.describe(backup_id="Backup-ID", dry_run="Nur Vorschau (kein Restore)")
@app_commands.default_permissions(administrator=True)
async def backup_restore(interaction: discord.Interaction, backup_id: str, dry_run: bool = True) -> None:
    await interaction.response.defer(thinking=True)
    backup = await bot.db.aget_backup(backup_id)
    if not backup:
        await interaction.followup.send(embed=create_embed(f"{E.FAIL} Nicht gefunden",f"Kein Backup `{backup_id}`",COLOR_DANGER)); return
    data = backup.get("data", {})
    info = (f"**Original:** {backup.get('guild_name','?')}\n**Rollen:** {len(data.get('roles',[]))}\n"
            f"**Text-Kanäle:** {len(data.get('text_channels',[]))}\n**Voice:** {len(data.get('voice_channels',[]))}")
    if dry_run:
        await interaction.followup.send(embed=create_embed(f"{E.BACKUP} Backup-Vorschau",info+"\n\n⚠️ Dry-Run – setze `dry_run:Nein` zum Wiederherstellen.",COLOR_INFO)); return
    restored = {"roles":0,"text_channels":0,"voice_channels":0,"errors":0}
    existing_names = {r.name for r in interaction.guild.roles}
    for rd in sorted(data.get("roles",[]), key=lambda r: r.get("position",0)):
        if rd["name"] in existing_names: continue
        try:
            await rate_limited(interaction.guild.create_role, name=rd["name"], color=discord.Color(rd.get("color",0)),
                               hoist=rd.get("hoist",False), mentionable=rd.get("mentionable",False),
                               permissions=discord.Permissions(rd.get("permissions",0)))
            restored["roles"] += 1
        except: restored["errors"] += 1
    existing_ch = {c.name for c in interaction.guild.channels}
    for cd in sorted(data.get("text_channels",[]), key=lambda c: c.get("position",0)):
        if cd["name"] in existing_ch: continue
        cat = discord.utils.get(interaction.guild.categories, name=cd.get("category_name")) if cd.get("category_name") else None
        try:
            await rate_limited(interaction.guild.create_text_channel, cd["name"], category=cat, topic=cd.get("topic"),
                               nsfw=cd.get("nsfw",False), slowmode_delay=cd.get("slowmode_delay",0))
            restored["text_channels"] += 1
        except: restored["errors"] += 1
    await interaction.followup.send(embed=create_embed(f"{E.OK} Backup wiederhergestellt",
        f"**Rollen:** {restored['roles']} | **Kanäle:** {restored['text_channels']} | **Fehler:** {restored['errors']}",COLOR_SUCCESS))
    await bot.db.aadd_admin_audit(interaction.guild.id,interaction.user.id,str(interaction.user),"backup_restore","backup",f"ID:{backup_id}")

    await interaction.response.defer(thinking=True, ephemeral=True)
    cases = await bot.db.aget_user_cases(interaction.guild.id, member.id, 500) if member else await bot.db.aget_recent_cases(interaction.guild.id, 500)
    if not cases:
        await interaction.followup.send(embed=create_embed(f"{E.FAIL} Keine Cases","",COLOR_INFO), ephemeral=True); return
    if format.value == "csv":
        out = StringIO(); writer = csv.writer(out)
        writer.writerow(["case_id","user_id","mod_id","action","reason","duration","created_at"])
        for c in cases:
            created = c.get("created_at","")
            if isinstance(created,datetime.datetime): created = created.isoformat()
            writer.writerow([c.get("case_id"),c.get("user_id"),c.get("mod_id"),c.get("action"),c.get("reason","")[:200],c.get("duration",""),created])
        fc = out.getvalue().encode("utf-8"); fn = f"cases_{interaction.guild.id}.csv"
    else:
        export = [{"case_id":c.get("case_id"),"user_id":c.get("user_id"),"mod_id":c.get("mod_id"),
                   "action":c.get("action"),"reason":c.get("reason",""),
                   "created_at":c["created_at"].isoformat() if isinstance(c.get("created_at"),datetime.datetime) else ""} for c in cases]
        fc = json.dumps(export, ensure_ascii=False, indent=2).encode("utf-8"); fn = f"cases_{interaction.guild.id}.json"
    file = discord.File(_io.BytesIO(fc), filename=fn)
    await interaction.followup.send(embed=create_embed(f"{E.OK} {len(cases)} Cases exportiert","",COLOR_SUCCESS), file=file, ephemeral=True)

@bot.tree.command(name="audit-perms", description="Permissions-Audit")
@app_commands.default_permissions(administrator=True)
async def slash_audit_perms(interaction: discord.Interaction) -> None:
    await interaction.response.defer(ephemeral=True, thinking=True)
    guild = interaction.guild
    cfg = bot.db.get_config(guild.id)
    danger_perms = cfg.get("perms_audit",{}).get("danger_perms",["administrator","manage_guild","manage_roles","manage_channels","ban_members","kick_members","mention_everyone","manage_webhooks","manage_messages","moderate_members"])
    findings: List[Tuple[str,str,str]] = []
    total_roles = max(len(guild.roles)-1,1)
    for role in sorted(guild.roles, key=lambda r: r.position, reverse=True):
        if role.is_default():
            risky = [p for p in danger_perms if getattr(role.permissions,p,False)]
            if risky: findings.append(("CRITICAL","@everyone",f"@everyone hat **{', '.join(risky)}**!"))
            continue
        perms = role.permissions; risky = [p for p in danger_perms if getattr(perms,p,False)]
        member_count = len(role.members); position_pct = int((role.position/total_roles)*100)
        if perms.administrator:
            findings.append(("CRITICAL",role.name,f"`Administrator` aktiv. {member_count} Member. Position: {position_pct}%.")); continue
        if risky:
            sev = "HIGH" if any(p in risky for p in ("ban_members","manage_guild","manage_roles","manage_webhooks")) else "MEDIUM"
            findings.append((sev,role.name,f"**{', '.join(risky)}** · {member_count} Member · {position_pct}%"))
    bot_top = guild.me.top_role
    higher = [r for r in guild.roles if r.position > bot_top.position and not r.is_default()]
    if higher: findings.append(("HIGH","Bot-Hierarchie",f"Bot-Rolle ist #{bot_top.position}. **{len(higher)} Rollen darüber** – nicht moderierbar."))
    sev_emoji = {"CRITICAL":"🔴","HIGH":"🟠","MEDIUM":"🟡","LOW":"🔵"}
    sev_color = {"CRITICAL":COLOR_DANGER,"HIGH":COLOR_DANGER,"MEDIUM":COLOR_WARNING,"LOW":COLOR_INFO}

@bot.event
async def on_command_error(ctx: commands.Context, error: Exception) -> None:
    if isinstance(error, commands.MissingPermissions):
        await ctx.send(embed=create_embed(f"{E.FAIL} Keine Berechtigung","",COLOR_DANGER))
    elif isinstance(error, commands.MemberNotFound):
        await ctx.send(embed=create_embed(f"{E.FAIL} Nutzer nicht gefunden","",COLOR_DANGER))
    elif isinstance(error, commands.CommandNotFound): return
    else: log.error(f"Command-Fehler: {error}")

@bot.tree.error
async def on_app_command_error(interaction: discord.Interaction, error: app_commands.AppCommandError) -> None:
    cmd_name = interaction.command.qualified_name if interaction.command else "?"
    user_msg: Optional[str] = None
    color = COLOR_DANGER
    if isinstance(error, app_commands.CommandOnCooldown):
        user_msg = f"{E.CLOCK} Warte noch **{error.retry_after:.1f}s**."; color = COLOR_WARNING
    elif isinstance(error, app_commands.MissingPermissions):
        user_msg = f"{E.FAIL} Dir fehlen: `{', '.join(error.missing_permissions)}`"
    elif isinstance(error, app_commands.BotMissingPermissions):
        user_msg = f"{E.FAIL} Dem Bot fehlen: `{', '.join(error.missing_permissions)}`"
    elif isinstance(error, (app_commands.CheckFailure, app_commands.CommandNotFound)):
        user_msg = f"{E.FAIL} Kein Zugriff."
    embed = create_embed(f"{E.FAIL} Fehler", user_msg or f"Unerwarteter Fehler: `{str(error)[:200]}`", color)
    try:
        if interaction.response.is_done(): await interaction.followup.send(embed=embed, ephemeral=True)
        else: await interaction.response.send_message(embed=embed, ephemeral=True)
    except: pass
    if user_msg is None:
        original = getattr(error,"original",error)
        tb = "".join(traceback.format_exception(type(original),original,original.__traceback__))
        log.error(f"Slash-Cmd '{cmd_name}': {tb[-1000:]}")

_LOGIN_ATTEMPTS: Dict[str, deque] = defaultdict(deque)
_LOGIN_LOCK = threading.Lock()
LOGIN_WINDOW = 300
LOGIN_MAX_TRIES = 7

def _client_ip() -> str:
    return (request.headers.get("X-Forwarded-For", request.remote_addr or "?")).split(",")[0].strip()

def _rate_limited_web(ip: str) -> bool:
    now = time.time()
    with _LOGIN_LOCK:
        dq = _LOGIN_ATTEMPTS[ip]
        while dq and now - dq[0] > LOGIN_WINDOW: dq.popleft()
        return len(dq) >= LOGIN_MAX_TRIES

def _record_attempt(ip: str) -> None:
    with _LOGIN_LOCK: _LOGIN_ATTEMPTS[ip].append(time.time())

def _run_async(coro, timeout: float = 8.0):
    if BOT_REF is None or not BOT_REF.loop or not BOT_REF.loop.is_running(): return None
    try: return asyncio.run_coroutine_threadsafe(coro, BOT_REF.loop).result(timeout=timeout)
    except Exception as e: log.debug(f"_run_async: {e}"); return None

def admin_required(fn):
    @wraps(fn)
    def wrapper(*args, **kwargs):
        if not session.get("admin"): return redirect(url_for("admin_login"))
        return fn(*args, **kwargs)
    return wrapper

def dashboard_login_required(fn):
    @wraps(fn)
    def wrapper(*args, **kwargs):
        if not session.get("discord_user"): return redirect(url_for("login"))
        return fn(*args, **kwargs)
    return wrapper

def _discord_api_get(endpoint: str, access_token: str) -> Optional[Any]:
    if not _REQUESTS_AVAILABLE: return None
    try:
        resp = _requests.get(f"{DISCORD_API_BASE}/{endpoint}",
                             headers={"Authorization": f"Bearer {access_token}"}, timeout=10)
        return resp.json() if resp.status_code == 200 else None
    except: return None

def _check_guild_access(guild_id: str) -> bool:
    guilds = session.get("discord_guilds", [])
    for g in guilds:
        if g["id"] == guild_id:
            return bool(int(g.get("permissions", 0)) & DISCORD_PERMISSIONS_ADMIN)
    return False

let currentCat = '';
function filterCmds(){
    const q = document.getElementById('cmdSearch').value.toLowerCase();
    const cards = document.querySelectorAll('.cmd-card');
    let visible = 0;
    cards.forEach(card=>{
        const matchQ = !q || card.dataset.name.includes(q) || card.dataset.desc.includes(q);
        const matchCat = !currentCat || card.dataset.cat === currentCat;

@app_flask.route("/auth/callback")
def auth_callback():
    """Discord OAuth2 Callback — tauscht Code gegen Token und holt User-Daten."""
    code  = request.args.get("code")
    state = request.args.get("state")
    error = request.args.get("error")

    # Error from Discord (e.g. user denied)
    if error:
        return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                      oauth_url=url_for("login"), error=f"Discord-Fehler: {error}")

    # State-Prüfung gegen Session
    if not state or state != session.get("oauth_state"):
        return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                      oauth_url=url_for("login"), error="Ungültiger OAuth2-State. Bitte versuche es erneut.")

    if not code:
        return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                      oauth_url=url_for("login"), error="Kein Authorization-Code erhalten.")

    if not _REQUESTS_AVAILABLE:
        return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                      oauth_url=url_for("login"), error="requests-Bibliothek nicht installiert.")

    try:
        # Token Exchange
        token_resp = _requests.post(
            "https://discord.com/api/oauth2/token",
            data={
                "client_id":     DISCORD_CLIENT_ID,
                "client_secret": DISCORD_CLIENT_SECRET,
                "grant_type":    "authorization_code",
                "code":          code,
                "redirect_uri":  DISCORD_REDIRECT_URI,
            },
            headers={"Content-Type": "application/x-www-form-urlencoded"},
            timeout=15
        )
        if token_resp.status_code != 200:
            log.error(f"OAuth2 token error: {token_resp.status_code} {token_resp.text[:200]}")
            return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                          oauth_url=url_for("login"), error=f"Token-Fehler ({token_resp.status_code}). Prüfe DISCORD_REDIRECT_URI.")
        token_data = token_resp.json()
        access_token = token_data.get("access_token")
        if not access_token:
            return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                          oauth_url=url_for("login"), error="Kein Access-Token erhalten.")

        # Fetch user info
        user_data = _requests.get("https://discord.com/api/users/@me",
                                   headers={"Authorization": f"Bearer {access_token}"}, timeout=10).json()
        if "id" not in user_data:
            return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                          oauth_url=url_for("login"), error="Nutzer-Daten konnten nicht abgerufen werden.")

        # Fetch guilds
        guilds_resp = _requests.get("https://discord.com/api/users/@me/guilds",
                                     headers={"Authorization": f"Bearer {access_token}"}, timeout=10)
        guilds_data = guilds_resp.json() if guilds_resp.status_code == 200 else []

        # Save to session
        session.permanent = True
        session["discord_user"] = {
            "id":            user_data.get("id"),
            "username":      user_data.get("username", "?"),
            "discriminator": user_data.get("discriminator", "0"),
            "avatar":        user_data.get("avatar"),
        }
        session["discord_guilds"]  = guilds_data
        session.pop("oauth_state", None)

        ACTIVITY.push("dashboard_login", f"Dashboard-Login: {user_data.get('username','?')}")
        return redirect(url_for("dashboard_index"))

    except Exception as e:
        log.error(f"OAuth2 callback exception: {e}")
        return render_template_string(DISCORD_LOGIN_PAGE, client_id=DISCORD_CLIENT_ID,
                                      oauth_url=url_for("login"), error=f"Fehler beim Login: {str(e)[:100]}")

@app_flask.route("/guild/<guild_id>/api/config", methods=["GET"])
def guild_api_config(guild_id: str):
    # Supports both session auth and API token auth
    auth_header = request.headers.get("Authorization","")
    if auth_header.startswith("Bearer "):
        token = auth_header[7:]
        if BOT_REF:
            token_doc = _run_async(BOT_REF.db.avalidate_api_token(token))
            if not token_doc: return jsonify({"error":"Ungültiger Token"}),401
        else: return jsonify({"error":"Bot nicht verfügbar"}),503
    elif not session.get("discord_user") or not _check_guild_access(guild_id):
        return jsonify({"error":"Nicht authentifiziert"}),401
    try: gid=int(guild_id)
    except: return jsonify({"error":"Ungültig"}),400
    if BOT_REF is None: return jsonify({"error":"Bot nicht verfügbar"}),503
    cfg=_run_async(BOT_REF.db.aget_config(gid))
    if cfg is None: return jsonify({"error":"Nicht gefunden"}),404
    return jsonify({k:v for k,v in cfg.items() if k!="_id"})

@app_flask.route("/guild/<guild_id>/api/module/<module>", methods=["POST"])
@dashboard_login_required
def guild_api_module(guild_id: str, module: str):
    if not _check_guild_access(guild_id):
        return jsonify({"error":"Kein Zugriff"}),403
    try: gid=int(guild_id)
    except: return jsonify({"error":"Ungültig"}),400
    if BOT_REF is None: return jsonify({"error":"Bot nicht verfügbar"}),503
    data=request.get_json()
    if not data: return jsonify({"error":"Kein JSON"}),400
    cfg=_run_async(BOT_REF.db.aget_config(gid))
    if cfg is None: return jsonify({"error":"Nicht gefunden"}),404
    if module in cfg and isinstance(cfg[module],dict):
        for k,v in data.items(): cfg[module][k]=v
        _run_async(BOT_REF.db.set_config(gid,cfg))
        usr=session.get("discord_user",{})
        _run_async(BOT_REF.db.aadd_admin_audit(gid,int(usr.get("id",0)),usr.get("username","?"),f"api_module_{module}",module,f"API: {list(data.keys())}"))
        return jsonify({"ok":True,"module":module})
    return jsonify({"error":f"Modul '{module}' nicht gefunden"}),404

        return jsonify({"error":"Kein Zugriff"}),403
    try: gid=int(guild_id)
    except: return jsonify({"error":"Ungültig"}),400
    if BOT_REF is None: return jsonify({"backups":[]})
    backups=_run_async(BOT_REF.db.aget_guild_backups(gid)) or []
